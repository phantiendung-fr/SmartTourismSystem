backend/
│
├── main.py                 # File chạy chính của server, khởi tạo các route.
├── database.py             # Logic kết nối SQL (Tìm hiểu cách kết nối db ở đây).
├── models.py               # Định nghĩa các bảng SQL (Users, Locations, Trips,...).
├── schemas.py              # Định nghĩa dữ liệu đầu vào/đầu ra (Ràng buộc điều khoản đăng nhập/đăng ký).
│
├── api/                    # Chứa các file xử lý API (Controllers)
│   ├── auth.py             # Xử lý Đăng nhập, Đăng ký, Đăng xuất, Phân quyền.
│   ├── locations.py        # Chức năng xử lí đầu vào & Gợi ý địa điểm.
│   └── trips.py            # Chức năng planning, tracking, tối ưu lộ trình, lưu lịch sử.
│
├── core/                   # Các logic cốt lõi và bảo mật
│   ├── security.py         # Mã hóa mật khẩu (bắt buộc nên làm, dùng thư viện passlib/bcrypt), tạo token phân quyền.
│   └── algorithms.py       # Thuật toán gợi ý địa điểm và tối ưu lộ trình (TSP).
│
├── crud/                   # (Create, Read, Update, Delete) - Chứa các câu Query SQL
│   ├── crud_user.py        # Query lấy thông tin user, kiểm tra đăng nhập.
│   └── crud_trip.py        # Query lấy lộ trình, lịch sử.
│
└── dummy_data/             # Thư mục chứa script tạo dữ liệu mẫu
    └── init_db.sql         # File SQL insert sẵn các địa điểm du lịch, user test.
